class OptionHeader(dict):

    def register(self, tag):
        return functools.partial(self._register, tag)

    def _register(self, tag, function):
        if tag in self:
            raise KeyError('Key {!r} conflict with existing item {}'.format(
                tag, self[tag]))
        self[tag] = function
        return function
